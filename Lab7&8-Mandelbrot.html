Introduction

This project visualizes the deep connection between the Mandelbrot set and Julia sets using Python and Flask. The Mandelbrot set serves as a map of complex numbers \( c \) that determines the behavior of the corresponding Julia set generated by the iterative function \( z_{n+1} = z_n^2 + c \). For each point \( c \) in the Mandelbrot set, the corresponding Julia set is connected; if \( c \) lies outside the Mandelbrot set, the Julia set is totally disconnected. By allowing users to input any complex number \( c \), the code dynamically plots the Mandelbrot set with that point marked in red and generates the associated Julia set, visually demonstrating how the global structure of the Mandelbrot set encodes the local behavior of Julia sets.

2. The Main Code Behind & Analysis

This Python code uses Flask to create a web application that dynamically generates and returns a PNG image displaying both the Mandelbrot set and its corresponding Julia set for any complex number \( c \) entered by the user. It employs Numba's @jit decorator to optimize performance of the iterative calculations used to determine pixel values in the Mandelbrot and Julia sets. The /mandelbrot route receives a complex number from the user via a URL parameter, computes both fractals using NumPy arrays, and plots them side by side with Matplotlib. The user’s point is marked in red on the Mandelbrot set to show its location, and the associated Julia set is plotted below. The final figure is streamed as a PNG image to the browser. The root route / simply serves the HTML frontend, and the server is configured to run on a dynamic port compatible with deployment platforms like Fly.io or Render. 


# Import necessary libraries
import numpy as np                       # Numerical operations and array handling
import matplotlib.pyplot as plt          # Plotting library for visualizing fractals
from numba import jit                    # Just-In-Time compiler for speed optimization
from flask import Flask, request, send_file, render_template  # Flask web framework tools
import io                                # For in-memory byte stream
import os                                # For accessing environment variables (e.g., PORT)

# Initialize the Flask app
app = Flask(__name__)

# -----------------------------------------
# Fractal Computation Functions
# -----------------------------------------

# Compute the number of iterations before escape for Mandelbrot formula
@jit(nopython=True)
def mandelbrot(c, max_iter):
    z = 0
    n = 0
    while abs(z) <= 2 and n < max_iter:
        z = z*z + c
        n += 1
    return n  # Return the number of iterations

# Generate the Mandelbrot set over a 2D grid
@jit(nopython=True)
def mandelbrot_set(xmin, xmax, ymin, ymax, width, height, max_iter):
    r1 = np.linspace(xmin, xmax, width)    # Real axis values
    r2 = np.linspace(ymin, ymax, height)   # Imaginary axis values
    n3 = np.empty((width, height))         # Array to store iteration counts

    for i in range(width):
        for j in range(height):
            # Combine real and imaginary parts to create complex numbers
            n3[i, j] = mandelbrot(r1[i] + 1j * r2[j], max_iter)
    return n3

# Generate the Julia set for a specific complex parameter c
@jit(nopython=True)
def julia_set(c, xmin, xmax, ymin, ymax, width, height, max_iter):
    r1 = np.linspace(xmin, xmax, width)
    r2 = np.linspace(ymin, ymax, height)
    n3 = np.empty((width, height))

    for i in range(width):
        for j in range(height):
            z = r1[i] + 1j * r2[j]  # Initialize z at each point
            n = 0
            while abs(z) <= 2 and n < max_iter:
                z = z*z + c         # Julia iteration
                n += 1
            n3[i, j] = n
    return n3

# -----------------------------------------
# Plotting Function
# -----------------------------------------

# Generate and plot Mandelbrot and Julia sets for a given complex number
def plot_both_sets(user_point):
    # Plot and image generation settings
    width, height = 800, 800
    max_iter = 800

    # Define view window for Mandelbrot set
    mandel_xmin, mandel_xmax, mandel_ymin, mandel_ymax = -1.5, 0.5, -1, 1
    mandelbrot_data = mandelbrot_set(mandel_xmin, mandel_xmax, mandel_ymin, mandel_ymax, width, height, max_iter)

    # Define view window for Julia set
    julia_xmin, julia_xmax, julia_ymin, julia_ymax = -1.5, 1.5, -1.5, 1.5
    julia_data = julia_set(user_point, julia_xmin, julia_xmax, julia_ymin, julia_ymax, width, height, 256)

    # Create figure with two vertically stacked subplots
    fig, axs = plt.subplots(2, 1, figsize=(10, 20))

    # Plot Mandelbrot set
    axs[0].imshow(mandelbrot_data.T, extent=[mandel_xmin, mandel_xmax, mandel_ymin, mandel_ymax], cmap='hot')
    axs[0].set_title("Mandelbrot Set")
    axs[0].set_xlabel("Re")
    axs[0].set_ylabel("Im")
    axs[0].plot(user_point.real, user_point.imag, 'ro', markersize=10)  # Highlight selected complex number
    axs[0].grid(True)

    # Plot Julia set
    axs[1].imshow(julia_data.T, extent=[julia_xmin, julia_xmax, julia_ymin, julia_ymax], cmap='hot', interpolation='bilinear')
    axs[1].set_title(f"Julia Set for c = {user_point}")
    axs[1].set_xlabel("Re")
    axs[1].set_ylabel("Im")
    axs[1].grid(True)

    # Finalize layout and export plot to a byte buffer
    plt.tight_layout()
    buf = io.BytesIO()
    plt.savefig(buf, format='png')
    plt.close(fig)
    buf.seek(0)
    return buf  # Return the PNG image in memory

# -----------------------------------------
# Web App Routes
# -----------------------------------------

# Root route: render HTML form for input
@app.route('/')
def index():
    return render_template('index.html')  # Loads from templates/index.html

# Image route: respond with Mandelbrot + Julia plot image
@app.route('/mandelbrot')
def mandelbrot_img():
    try:
        # Get complex number from URL query (e.g., ?c=0.3+0.4j)
        c = complex(request.args.get('c', '0+0j'))
    except ValueError:
        # Fallback to 0 if input is invalid
        c = 0 + 0j
    # Generate and return the image
    return send_file(plot_both_sets(c), mimetype='image/png')

# Run the app if this file is the entry point
if __name__ == '__main__':
    port = int(os.environ.get('PORT', 5000))  # Use PORT env var if set, else default to 5000
    app.run(host='0.0.0.0', port=port)        # Bind to all IPs (required for deployment)
